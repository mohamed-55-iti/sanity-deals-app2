    <script>
        // 🔐 رابط Backend - غيّره بعد رفع Backend على Railway
        const API_URL = 'https://your-backend.railway.app';
        
        let currentDeal = null;
        let stores = [];

        // تحميل المطاعم عند فتح التطبيق
        window.addEventListener('DOMContentLoaded', async () => {
            await loadStores();
            loadCookiesFromStorage();
        });

        // جلب المطاعم
        async function loadStores() {
            try {
                showStatus('جاري تحميل المطاعم...', 'info');
                
                const response = await fetch(`${API_URL}/api/fetch-stores`);
                const result = await response.json();

                if (result.success) {
                    stores = result.stores;
                    
                    const select = document.getElementById('store');
                    select.innerHTML = '<option value="">-- اختر مطعم --</option>';
                    stores.forEach(store => {
                        const option = document.createElement('option');
                        option.value = store._id;
                        option.textContent = store.name;
                        select.appendChild(option);
                    });
                    
                    hideStatus();
                } else {
                    showStatus('❌ فشل تحميل المطاعم', 'error');
                }
            } catch (error) {
                showStatus('❌ خطأ: ' + error.message, 'error');
            }
        }

        // استخراج العرض
        async function extractDeal() {
            const url = document.getElementById('fbUrl').value.trim();
            
            if (!url) {
                showStatus('⚠️ أدخل رابط فيسبوك', 'error');
                return;
            }

            try {
                showLoading(true);
                hideStatus();

                const cookies = getCookies();

                const response = await fetch(`${API_URL}/api/extract-fb-data`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ url, cookies })
                });

                const result = await response.json();
                showLoading(false);

                if (result.success) {
                    currentDeal = result.data;
                    
                    document.getElementById('name_ar').value = result.data.name_ar;
                    document.getElementById('name_en').value = result.data.name_en;
                    document.getElementById('description_ar').value = result.data.description_ar;
                    document.getElementById('description_en').value = result.data.description_en;
                    document.getElementById('price').value = result.data.price;
                    document.getElementById('validFrom').value = result.data.validFrom;
                    document.getElementById('validTo').value = result.data.validTo;

                    if (result.data.image_url) {
                        document.getElementById('dealImage').src = result.data.image_url;
                        document.getElementById('imagePreview').classList.remove('hidden');
                    }

                    document.getElementById('dealForm').classList.remove('hidden');
                    showStatus('✅ تم الاستخراج بنجاح!', 'success');
                } else {
                    showStatus('❌ فشل: ' + result.error, 'error');
                }
            } catch (error) {
                showLoading(false);
                showStatus('❌ خطأ: ' + error.message, 'error');
            }
        }

        // رفع العرض
        async function uploadDeal() {
            try {
                const storeId = document.getElementById('store').value;
                
                if (!storeId) {
                    showStatus('⚠️ اختر المطعم', 'error');
                    return;
                }

                showLoading(true);

                // 1. رفع الصورة
                let imageAssetId = null;
                if (currentDeal.image_base64) {
                    const imgResponse = await fetch(`${API_URL}/api/upload-image`, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify({ imageBase64: currentDeal.image_base64 })
                    });
                    const imgResult = await imgResponse.json();
                    if (imgResult.success) {
                        imageAssetId = imgResult.assetId;
                    }
                }

                // 2. رفع العرض
                const dealData = {
                    name_ar: document.getElementById('name_ar').value,
                    name_en: document.getElementById('name_en').value,
                    description_ar: document.getElementById('description_ar').value,
                    description_en: document.getElementById('description_en').value,
                    price: document.getElementById('price').value,
                    slug: currentDeal.slug,
                    validFrom: document.getElementById('validFrom').value,
                    validTo: document.getElementById('validTo').value,
                    store: storeId,
                    imageAssetId: imageAssetId
                };

                const response = await fetch(`${API_URL}/api/upload-deal`, {
                    method: 'POST',
                    headers: { 'Content-Type': 'application/json' },
                    body: JSON.stringify({ deal: dealData })
                });

                const result = await response.json();
                showLoading(false);

                if (result.success) {
                    showStatus('✅ تم رفع العرض بنجاح!', 'success');
                    setTimeout(() => {
                        document.getElementById('dealForm').classList.add('hidden');
                        document.getElementById('fbUrl').value = '';
                        currentDeal = null;
                    }, 2000);
                } else {
                    showStatus('❌ فشل الرفع: ' + result.error, 'error');
                }
            } catch (error) {
                showLoading(false);
                showStatus('❌ خطأ: ' + error.message, 'error');
            }
        }

        // حفظ/تحميل الكوكيز
        function getCookies() {
            const cookies = localStorage.getItem('fb_cookies');
            return cookies ? JSON.parse(cookies) : null;
        }

        function loadCookiesFromStorage() {
            // يتم تحميل الكوكيز تلقائياً
        }

        function showStatus(message, type) {
            const el = document.getElementById('statusMessage');
            el.textContent = message;
            el.className = `status ${type} active`;
        }

        function hideStatus() {
            document.getElementById('statusMessage').classList.remove('active');
        }

        function showLoading(show) {
            document.getElementById('loading').classList.toggle('active', show);
        }
